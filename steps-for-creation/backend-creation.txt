STEPS TO CONNECT TO MONGODB USING MONGOOSE
1.start the MongoDB compass and connect to database
2.start the mongo shell and enter when shell opens
3.start the mongodb server using mongod command


const express = require("express");
const mongoose = require("mongoose");
const app = express();
mongoose.connect("mongodb://localhost:27017/fruitsDB");

const fruitSchema = new mongoose.Schema({
    name:String,
    count:Number
});

const Fruit = mongoose.model("Fruit",fruitSchema);

const fruit = new Fruit({
    name:"banana",
    count:5
});

fruit.save();


app.get("/",function(request,response){
 
    Fruit.find({name:"banana"}).then((fruits)=>{
        response.send(fruits);
    })
})

app.listen(3000,()=>console.log("successfully listening to the port 3000"));
